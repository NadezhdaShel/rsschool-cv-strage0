Code example

Solution task “Walk the Object Path?” from website Codewars: 

function find(object, path) {
  let p = path.split('.');
  for (let j = 0; j < p.length; j++) {
    if (object === undefined || !object.hasOwnProperty(p[j])) { return undefined }
    object = object[p[j]];
  }
  return object;
}
